\begin{quote}
\$ make menuconfig
\end{quote}

\begin{quote}
\$ ./scripts/config --disable SYSTEM_TRUSTED_KEYS
\$ ./scripts/config --disable SYSTEM_REVOCATION_KEYS
\$ ./scripts/config --disable MODULE_SIG_KEY
\end{quote}
